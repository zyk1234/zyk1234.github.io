导入train.csv数据的问题

整体思路为：单进程读取数据，进程池来消费数据

import multiprocessing
import dbmysql
import time


def read(qq):
    with open("train2.csv", 'r', encoding='utf-8') as f:
        f.readline()
        while True:
            block = f.read(1024)  # 一次读取1k数据
            if not block:
                break
            lists = []  # 打算一次插入多条数据
            for item in block.split('\n'):
                data = item.split(',')
                if len(data) != 6:
                    continue
                lists.append((data[0], data[1], data[2], data[3], data[4], '%s' % data[4]))
            qq.put(lists)


def write(q):
    while 1:  # 因为有可能还没有读出来，而已经进程池已经写完了
        count = 0  # 累计队列为空的次数，如果超过10次了，让当前的进程推出，设置此值就是为了给循环设定个终止条件
        while not q.empty():
            value = q.get()
            sql = "insert into train(row_id,x,y,accuracy,time,place_id) values (%s,%s,%s,%s,%s,%s);"
            dbmysql.executeMany(sql, value)
        else:
            time.sleep(3)  # 队列为空的时候，让当前线程等待3秒中
            count += 1
        time.sleep(10)
        if count == 10:
            break


if __name__ == '__main__':
    qq = multiprocessing.Manager().Queue()
    # 单进程读取数据
    p2 = multiprocessing.Process(target=read, args=(qq,))
    p2.start()
    time.sleep(20)  # 20秒后进程池写入数据
    p = multiprocessing.Pool(4)
    p.apply_async(write, args=(qq,))

    p.close()
    p.join()
