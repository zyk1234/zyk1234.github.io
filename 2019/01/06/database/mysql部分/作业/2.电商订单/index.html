import userdao, util #依赖userdao.py和util.py


class User:
    def __init__(self):
        self.rs = None

    def register(self):
        phoneNo = input("请输入手机号")
        password = input("请输入密码")
        # 判断手机号是否存在，如果存在返回
        rs = userdao.getUser(phoneNo)
        if rs[0] > 0:
            print('此用户已存在')
            return
        else:
            pwd = util.getMd5(password)  # 1.防止sql注入，2.防止数据库被攻击后，用户信息丢失，对于个人的隐私信息，存入到数据库中，可以选择密文存储
            userdao.insertUser(phoneNo=phoneNo, password=pwd)

    def login(self):
        phoneNo = input("请输入手机号")
        password = input("请输入密码")
        pwd=util.getMd5(password)
        self.rs = userdao.getUserByPhoneNoAndPwd(phoneNo, pwd)
        if self.rs:
            print('登录成功')
            return 1
        else:
            print('登录失败')
            return 2

    def updatePwd(self):
        if self.rs == None:
            self.login()
        password = input("请输入新密码")
        pwd = util.getMd5(password)
        if userdao.modifyPassword(userId=self.rs[0], password=pwd)==True:
            print('修改成功')

import logging,uuid,hashlib

#配置日志
logging.basicConfig(level=logging.INFO, format='[%(asctime)s -%(filename)s- %(name)s - %(levelname)s] - %(message)s')
logger = logging.getLogger(__name__)

#获取uuid
def getuuid():
    return str(uuid.uuid4())

#获取md5密文
def getMd5(source):
    m1 = hashlib.md5()
    # 使用md5对象里的update方法md5转换
    m1.update(source.encode("utf-8"))
    hexd = m1.hexdigest()
    return hexd

import pymysql
import util

'''
操作mysql第一步：创建连接，返回连接对象和游标
'''

host = "127.0.0.1"  # 本地的ip地址
port = 33006  # 端口号
user = "root"  # 数据库的用户名
passwd = "123456"  # 数据库的密码
db = "ddd3"  # 你要连接的数据库名称


def getConnection():
    try:
        conn = pymysql.connect(host=host, port=port, user=user, passwd=passwd, charset="utf8", db=db)
        cur = conn.cursor()
        return conn, cur
    except Exception as e:
        util.logger.error(e)
        return None


'''
第二步：创建增删改的方法，此方式用于insert、update、delete,create table|database,alter,drop databse|table
'''


def execute(sql):
    # 获取连接
    conn, cur = getConnection()
    try:
        # 游标执行sql语句
        f = cur.execute(sql)
        # 连接进行事务提交
        conn.commit()
        # 如果程序执行无误，返回True
        return True
    except Exception as e:
        util.logger.error(e)
        return False
    finally:
        cur.close()
        conn.close()


# 事务的
def execute2(sql):
    # 获取连接
    conn, cur = getConnection()
    try:
        conn.begin()
        # 游标执行sql语句
        for item in sql:
            cur.execute(item)
        # 连接进行事务提交
        conn.commit()
        # 如果程序执行无误，返回True
        return True
    except Exception as e:
        util.logger.error(e)
        return False
    finally:
        cur.close()
        conn.close()
